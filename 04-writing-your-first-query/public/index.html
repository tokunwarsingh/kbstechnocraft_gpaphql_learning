GraphQL Query Examples

This page demonstrates different types of GraphQL queries using the JSON data.

1. Simple Query - Get All Continents
query {
  continents {
    code
    name
  }
}
2. Query with Arguments - Get Specific Continent
query {
  continent(code: "AF") {
    name
    code
  }
}
3. Nested Query - Continents with Countries
query {
  continents {
    name
    countries {
      name
      capital
    }
  }
}
4. Query with Aliases - Multiple Continents
query {
  africa: continent(code: "AF") {
    name
  }
  europe: continent(code: "EU") {
    name
  }
}
5. Fragment - Reusable Fields
fragment CountryFields on Country {
  name
  capital
  currency
}

query {
  countries {
    ...CountryFields
  }
}
6. Query with Variables
query GetContinent($code: String!) {
  continent(code: $code) {
    name
    countries {
      name
    }
  }
}

Variables: {"code": "AS"}

7. Query with Directives
query GetContinents($includeCountries: Boolean!) {
  continents {
    name
    countries @include(if: $includeCountries) {
      name
    }
  }
}

Variables: {"includeCountries": true}

8. Get All Countries
query {
  countries {
    code
    name
    capital
  }
}
9. Get Country with States
query {
  country(code: "US") {
    name
    states {
      name
    }
  }
}
10. Get Languages
query {
  languages {
    code
    name
    native
  }
}
11. Get States for a Country
query {
  states(countryCode: "IN") {
    code
    name
  }
}
